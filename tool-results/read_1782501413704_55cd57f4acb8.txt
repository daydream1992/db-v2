    62→    62→def parse_meta(script_path: Path) -> dict:
    63→    63→    """从脚本头部解析 @meta 元数据"""
    64→    64→    try:
    65→    65→        content = script_path.read_text(encoding='utf-8')
    66→    66→    except Exception:
    67→    67→        return {}
    68→    68→
    69→    69→    meta = {}
    70→    70→    # 解析一行中的多个 # @meta key=value
    71→    71→    for line in content.split('\n'):
    72→    72→        # 匹配整行：# @meta key=value key=value ...
    73→    73→        m = re.match(r'#\s*@meta\s+(.*)', line)
    74→    74→        if m:
    75→    75→            # 解析所有 key=value
    76→    76→            for kv in m.group(1).split():
    77→    77→                parts = kv.split('=', 1)
    78→    78→                if len(parts) == 2:
    79→    79→                    meta[parts[0].strip()] = parts[1].strip()
    80→    80→    return meta

   138→   138→def filter_by_schedule(scripts_meta: dict, schedules: list) -> list:
   139→   139→    """按schedule过滤，先1_入库再2_计算，按sort排"""

   143→   143→                 if meta.get('_dir') == d and meta.get('schedule', 'once') in schedules]

   149→   149→def match_keyword(scripts_meta: dict, keyword: str) -> list:
   150→   150→    """按关键字匹配表名/中文名/sort编号"""
   151→   151→    kw = keyword.lower()
   152→   152→    result = []
   153→   153→    for d in DIR_ORDER:
   154→   154→        items = []
   155→   155→        for name, meta in scripts_meta.items():
   156→   156→            if meta.get('_dir') != d:
   157→   157→                continue
   158→   158→            sort_raw = meta.get('sort', '')
   159→   159→            sort = str(int(sort_raw)) if sort_raw else ''
   160→   160→            if (name == kw or kw in name
   161→   161→                    or kw in meta.get('cn', '').lower()
   162→   162→                    or sort.startswith(kw)):
   163→   163→                items.append((name, int(meta.get('sort', 999))))
   164→   164→        items.sort(key=lambda x: x[1])
   165→   165→        result.extend([n for n, _ in items])
   166→   166→    return list(dict.fromkeys(result))  # 去重保序

   353→   353→def _normalize_datestr(val) -> str:
   354→   354→    """把任何日期值统一成 'YYYY-MM-DD'，保证 _freshness 词法比较正确。
   355→   355→
   356→   356→    DuckDB DATE 列返回 datetime.date → str 已是 ISO; 但快照表的 HqDate 是
   357→   357→    VARCHAR 存 'YYYYMMDD' 紧凑串, 不规范会误判新鲜度。
   358→   358→    """
   359→   359→    if val is None:
   360→   360→        return ''
   361→   361→    s = str(val)
   362→   362→    if len(s) >= 10 and s[4] in '-/':
   363→   363→        return s[:10]            # 已是 ISO / 带分隔符
   364→   364→    if len(s) == 8 and s.isdigit():
   365→   365→        return f'{s[0:4]}-{s[4:6]}-{s[6:8]}'   # YYYYMMDD → YYYY-MM-DD
   366→   366→    return s[:10]

   388→   388→def _last_trading_day(con) -> object:
   389→   389→    """最近一个交易日(is_trading=1 且 <= 今天); 查不到回退到跳过周末的启发式。"""
   390→   390→    try:
   391→   391→        r = con.execute(
   392→   392→            "SELECT MAX(date) FROM trading_calendar WHERE is_trading=1 AND date <= CURRENT_DATE"
   393→   393→        ).fetchone()
   394→   394→        if r and r[0]:
   395→   395→            return r[0]
   396→   396→    except Exception:
   397→   397→        pass
   398→   398→    from datetime import timedelta
   399→   399→    d = datetime.now().date()
   400→   400→    while d.weekday() >= 5:  # 5=周六 6=周日
   401→   401→        d -= timedelta(days=1)
   402→   402→    return d

   417→   417→def _freshness(schedule: str, max_date, last_td, row_count: int) -> tuple:
   418→   418→    """按 schedule 判新鲜度，返回 (rich 颜色, 文案)。"""
   419→   419→    if row_count == 0:
   420→   420→        return ('red', '空表')
   421→   421→    sched = (schedule or '').lower()
   422→   422→    if sched in ('', 'once'):
   423→   423→        return ('white', '—')
   424→   424→    if max_date is None:
   425→   425→        return ('yellow', '无日期列')
   426→   426→    md = str(max_date)[:10]
   427→   427→    from datetime import timedelta
   428→   428→    if sched == 'daily':
   429→   429→        if last_td is not None and md < str(last_td)[:10]:
   430→   430→            return ('red', f'滞后{md}')

   432→   432→    if sched == 'weekly':
   433→   433→        cutoff = (datetime.now() - timedelta(days=7)).date()

   436→   436→        cutoff = (datetime.now() - timedelta(days=35)).date()

   456→   456→def _health_autofix(red_tables: list, scripts_meta: dict, yes: bool):
   457→   457→    """对 health 标红的表逐个重跑补数 (force=True)。
   458→   458→
   459→   459→    安全: 默认逐表 [y/N] 确认 (默认 N=跳过); --yes 跳过确认批量重跑。
   460→   460→    大表(>5千万行)额外提示耗时。属敏感操作, 故默认不自动执行。
   461→   461→    """
   462→   462→    if not red_tables:
   463→   463→        logger.info('无滞后/空表，无需补数')
   464→   464→        return
   465→   465→    BIG = 50_000_000
   466→   466→    logger.info(f'开始补数（{len(red_tables)} 张滞后/空表）')
   467→   467→    results = []
   468→   468→    for name, cnt in red_tables:
   469→   469→        if not yes:
   470→   470→            tag = f' [大表 {cnt:,} 行, 重跑耗时长]' if cnt >= BIG else f' [{cnt:,} 行]'
   471→   471→            ans = input(f'[fix?] {name}{tag} 重跑? [y/N]: ').strip().lower()
   472→   472→            if ans != 'y':
   473→   473→                logger.info(f'{name}: 跳过')
   474→   474→                results.append((name, 'SKIP'))
   475→   475→                continue
   476→   476→        logger.info(f'重跑 {name} (force=True) ...')
   477→   477→        ok = _rerun_table(name, scripts_meta)
   478→   478→        results.append((name, 'OK' if ok else 'FAIL'))
   479→   479→    _print_summary(results)

   563→   563→                   key=lambda x: (DIR_ORDER.index(x[1].get('_dir', '1_入库'))

   597→   597→def cmd_join(table_name: str):
   598→   598→    """关联发现 — 找该表的 _labeled 视图 + 可 join 的 dim 表 + 推荐 join key + JOIN 模板。
   599→   599→
   600→   600→    解决: 取数时不知道这张表能 join 谁、用什么 key。
   601→   601→    """
   602→   602→    from config.gen_data_dict import build_data_dict
   603→   603→    con = duckdb.connect(str(DB_PATH), read_only=True)
   604→   604→    db_tables = set(con.execute("SHOW TABLES").fetchdf()['name'].tolist())
   605→   605→    if table_name not in db_tables:
   606→   606→        logger.error(f'{table_name}: DB 无此表')
   607→   607→        con.close()
   608→   608→        return
   609→   609→
   610→   610→    try:
   611→   611→        target_cols = set(con.execute(f'DESCRIBE {table_name}').fetchdf()['column_name'].tolist())
   612→   612→    except Exception as e:
   613→   613→        logger.error(f'DESCRIBE {table_name} 失败: {e}')
   614→   614→        con.close()
   615→   615→        return
   616→   616→
   617→   617→    logger.info(f'目标表: {table_name}（{len(target_cols)} 列）')
   618→   618→
   619→   619→    # 1) _labeled 视图 (字段含义已 JOIN, 取数首选)
   620→   620→    labeled = f'{table_name}_labeled'
   621→   621→    if labeled in db_tables:
   622→   622→        logger.info(f'已有取数视图: {labeled}  (字段含义已 JOIN，直接查它即可)')
   623→   623→    else:
   624→   624→        logger.info(f'无 {labeled} 视图')
   625→   625→
   626→   626→    # 2) 候选 dim 表 (列名能交上的才有 join 可能)
   627→   627→    data_dict = build_data_dict()
   628→   628→    dim_tables = [t for t, v in data_dict.items()
   629→   629→                  if t != '_meta' and (v.get('is_dim') or t.startswith('dim_') or t.endswith('_indicator'))]
   630→   630→
   631→   631→    candidates = []
   632→   632→    for dt in dim_tables:
   633→   633→        try:
   634→   634→            dcols = set(con.execute(f'DESCRIBE {dt}').fetchdf()['column_name'].tolist())
   635→   635→        except Exception:
   636→   636→            continue
   637→   637→        shared = target_cols & dcols
   638→   638→        if not shared:
   639→   639→            continue
   640→   640→        key = next((k for k in _JOIN_KEY_PRIORITY if k in shared), sorted(shared)[0])
   641→   641→        candidates.append((dt, key, sorted(shared), data_dict.get(dt, {}).get('cn', '')))
   642→   642→
   643→   643→    if candidates:
   644→   644→        logger.info(f'可关联的维度表（{len(candidates)}）:')
   645→   645→        for dt, key, shared, cn in candidates:
   646→   646→            Console().print(f'   • [cyan]{dt:<30}[/cyan] join key=[green]{key}[/green]  共有列={shared}  {cn}')
   647→   647→        best_dt, best_key, _, _ = candidates[0]
   648→   648→        Console().print('\n   [dim]JOIN 模板（复制即用）:[/dim]')
   649→   649→        Console().print(f'   [green]SELECT t.*, d.*[/green]')
   650→   650→        Console().print(f'   [green]FROM {table_name} t[/green]')
   651→   651→        Console().print(f'   [green]LEFT JOIN {best_dt} d ON t.{best_key} = d.{best_key}[/green]')
   652→   652→    else:
   653→   653→        logger.warning('没有列名能匹配上的 dim 表（可能需先建 dim_*_indicator 维度表，或在脚本里加 FIELD_MAP）')
   654→   654→
   655→   655→    con.close()
