   716→// ═══════════════════════════════════════════════════════════
   717→// 健康度矩阵 — 基于数据库表扫描结果判定
   718→// 逻辑：扫描每张表的 rows + maxDate，对照交易日历，
   719→//       有数据覆盖→success，数据缺失→failed，非交易日→skipped，不适用→none
   720→// 扫描 SQL: SELECT COUNT(*) AS rows, MAX({date_col}) AS max_date FROM {table}
   721→// ═══════════════════════════════════════════════════════════
   722→
   723→/** 判断某表在某交易日是否有数据覆盖 — 基于数据库扫描结果 */
   724→function tableHasDataForDate(t: TableMeta, mmdd: string): 'success' | 'failed' | 'none' {
   725→  // 无日期列 且 schedule=once → 不适用每日数据覆盖检测
   726→  if (!t.dateCol && t.schedule === 'once') return 'none'
   727→
   728→  // 空表 → 缺数据
   729→  if (t.rows === 0) return 'failed'
   730→
   731→  // 无最大日期 → 缺数据（有行但无法判断日期覆盖范围）
   732→  if (!t.maxDate) return 'failed'
   733→
   734→  const fullDate = `2026-${mmdd}`
   735→  const maxDate = t.maxDate
   736→
   737→  if (t.schedule === 'daily') {
   738→    // 日频表：maxDate >= 交易日 → 有数据，否则缺数据
   739→    return maxDate >= fullDate ? 'success' : 'failed'
   740→  }
   741→
   742→  if (t.schedule === 'weekly') {
   743→    // 周频表：maxDate 应在 7 天内（当前周未结束，上周数据仍有效）
   744→    const diffMs = new Date(fullDate).getTime() - new Date(maxDate).getTime()
   745→    const diffDays = diffMs / 86400000
   746→    return diffDays <= 7 ? 'success' : 'failed'
   747→  }
   748→
   749→  if (t.schedule === 'monthly') {
   750→    // 月频表：maxDate 应在同一月或上一月（当月未结束，上月数据仍有效）
   751→    const maxM = new Date(maxDate).getFullYear() * 12 + new Date(maxDate).getMonth()
   752→    const checkM = new Date(fullDate).getFullYear() * 12 + new Date(fullDate).getMonth()
   753→    return maxM >= checkM - 1 ? 'success' : 'failed'
   754→  }
   755→
   756→  // once 但有日期列：有数据即可
   757→  return t.rows > 0 ? 'success' : 'failed'
   758→}

   760→/** 基于数据库扫描结果推导健康色 — 不依赖脚本执行日志 */
   761→export function deriveHealthFromScan(t: TableMeta): HealthColor {
   762→  // once 且无日期列 → 不适用
   763→  if (t.schedule === 'once' && !t.dateCol) return 'white'
   764→  // 空表 → 红
   765→  if (t.rows === 0) return 'red'
   766→  // 无最大日期 → 红
   767→  if (!t.maxDate) return 'red'
   768→
   769→  const lastTradingFull = `2026-${LAST_TRADING_DATE}`
   770→  const maxDate = t.maxDate
   771→
   772→  if (t.schedule === 'daily') {
   773→    if (maxDate >= lastTradingFull) return 'green'
   774→    const diffDays = Math.round((new Date(lastTradingFull).getTime() - new Date(maxDate).getTime()) / 86400000)
   775→    if (diffDays <= 2) return 'yellow'
   776→    return 'red'
   777→  }
   778→  if (t.schedule === 'weekly') {
   779→    const diffDays = Math.round((new Date(lastTradingFull).getTime() - new Date(maxDate).getTime()) / 86400000)
   780→    return diffDays <= 7 ? 'green' : 'yellow'
   781→  }
   782→  if (t.schedule === 'monthly') {
   783→    const maxM = new Date(maxDate).getFullYear() * 12 + new Date(maxDate).getMonth()
   784→    const checkM = new Date(lastTradingFull).getFullYear() * 12 + new Date(lastTradingFull).getMonth()
   785→    return maxM >= checkM - 1 ? 'green' : 'yellow'
   786→  }
   787→  return 'white'
   788→}

   791→export function getScanSQL(t: TableMeta): string {
   792→  if (!t.dateCol) return `SELECT COUNT(*) AS rows FROM ${t.table}`
   793→  return `SELECT COUNT(*) AS rows, MAX(${t.dateCol}) AS max_date FROM ${t.table}`
   794→}
